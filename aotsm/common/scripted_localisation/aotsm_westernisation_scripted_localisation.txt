#WESTERNISATION - COUNTRY_STATUS

defined_text = { 
	name = get_w_pru_country_status
	
	text = { 
		trigger = {
            PRU = {
                exists = yes
                has_opinion = {
					target = ROOT
					value < 30
				}
            }
            NOT = {
                ROOT = {
                     has_country_flag = aotsm_w_country_has_pru_protection 
                }
            }
        }
		localization_key = aotsm_w_country_exists_status_lk
	}
	
	text = { 
		trigger = {
			PRU = {
                exists = no
            }
        }
		localization_key = aotsm_w_country_not_exists_status_lk
	}
	
	text = { 
		trigger = {
			PRU = {
                exists = yes
                has_opinion = {
					target = ROOT
					value > 29
				}
            }
            NOT = {
                ROOT = {
                     has_country_flag = aotsm_w_country_has_pru_protection 
                }
            }
        }
		localization_key = aotsm_w_country_intrested_in_us_status_lk
	}
	
	
	text = { 
		trigger = { 
			PRU = {
               exists = yes
            }
            ROOT = { 
                has_country_flag = aotsm_w_country_has_pru_protection 
            }
        }
		localization_key = aotsm_w_country_protection_on_us_status_lk
	}
	
}
