	}

jap_invasion_on_kor = {
	
	allowed = {
		OR = {
			original_tag = JAP
			original_tag = JAS
			original_tag = TUK
		}
	}
	
	enable = {
		focus_progress = {
			focus = japan_korea_prepear
			progress > 0.01
		}
	}

	abort = {
		NOT = {
			country_exists = KOR
		}
	}

	ai_strategy = {
		type = front_control
		ordertype = invasion
        execute_order = yes
		tag = "KOR"
		priority = 10
	}
}
