bulgaria_war_preparation = {
	allowed = {
		original_tag = BYZ
	}
	enable = {
		focus_progress = {
			focus = BYZ_bulgaria_war
			progress > 0.01
		}
		NOT = {
			has_completed_focus = BYZ_bulgaria_war
			owns_state = 801
		}
		country_exists = BUL
	}

	abort = {
		OR = {
			NOT = { country_exists = BUL }

			AND = {
				has_completed_focus = BYZ_bulgaria_war
				BYZ = { has_war_with = BUL }
			}

			owns_state = 801
		}
	}

	ai_strategy = {
		type = conquer
		id = "BUL"
		value = 100
	}
}
