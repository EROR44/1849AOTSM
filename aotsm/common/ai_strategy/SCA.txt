sca_war_test_russian = {
	allowed = {
		original_tag = SCA
	}
	enable = {
		focus_progress = {
			focus = SCA_test_russian_army
			progress > 0.01
		}
		NOT = {
			has_completed_focus = SCA_test_russian_army
		}
		country_exists = RUS
	}

	abort = {
		OR = {
			NOT = { country_exists = RUS }
			has_completed_focus = SCA_test_russian_army
		}
	}

	ai_strategy = {
		type = conquer
		id = "RUS"
		value = 100
	}
}
