kur_dont_like_byzantium_war = {

	allowed = {
		tag = KUR
	}
	
	enable = {
		KUR = {
			OR = {
				is_puppet_of = BYZ
				is_subject_of = BYZ
			}
		}
	}
	
	abort = {
		NOT = {
			country_exists = BYZ
		}
	}
	
	ai_strategy = {
		type = diplo_action_acceptance
		id = BYZ
		value = -9999
		target = call_allies
	}
	
	ai_strategy = {
	    type = diplo_action_desire
	    id = BYZ
	    value = -9999
	    target = join_allies
	}
	
}
