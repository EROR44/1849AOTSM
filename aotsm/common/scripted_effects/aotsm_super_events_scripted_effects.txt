aotsm_se_test_effect = {
    if = {
        limit = { #проверка заблоканы суперивенты или нет
            NOT = { has_global_flag = aotsm_se_locked_flag }
        }
        set_global_flag = aotsm_se_visible_flag
        set_global_flag = aotsm_se_test_flag
    }  
}
