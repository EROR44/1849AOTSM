	PRU = {
		clr_country_flag = decaing_reich
	}
	set_global_flag = blacksun_civilwar_going
	every_state = {
		limit = {
			is_core_of = PRU
			is_owned_by = PRU
		}
		add_core_of = KGR
		add_core_of = RGR
		add_core_of = SGE
	}
	PRU = {
		country_event = ger_blackorder.50
	}
	hidden_effect = {
		PRU = {
			remove_mission = pru_start_of_civil_war_decision

		}
	}

}
