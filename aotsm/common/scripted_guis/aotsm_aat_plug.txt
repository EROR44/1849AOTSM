		context_type = player_context
		window_name = "topbar_aotsm_dip_button_plug"
		parent_window_token = top_bar
		
		visible = {
			always = yes
		}
		
		triggers = {
			aotsm_im_plug_button_pd_visible = {
				NOT = {
					has_country_flag = aotsm_politics_display_open
				}
			}
			aotsm_im_plug_button_pd_close_visible = {
				has_country_flag = aotsm_politics_display_open
			}
		}
		
		effects = {
			aotsm_im_plug_button_pd_click = {
				set_country_flag = aotsm_politics_display_open
			}
			aotsm_im_plug_button_pd_close_click = {
				clr_country_flag = aotsm_politics_display_open
			}
		}
		
	}
