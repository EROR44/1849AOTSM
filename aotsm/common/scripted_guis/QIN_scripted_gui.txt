scripted_gui = {

	QIN_selection_of_the_successor_scripted_gui = {
        context_type = decision_category
		window_name = "QIN_selection_of_the_successor_gui_window"

        effects = {
     		
        }

	}

	QIN_country_situation_scripted_gui = {
        context_type = decision_category
		window_name = "QIN_country_situation_scripted_gui"

        effects = {
     		QIN_open_region_leaders_list_button_click = {
                if = {
                    limit = {
                        ROOT = { NOT = { has_country_flag = QIN_visible_region_leaders_list_flag } }
                    }
                    ROOT = {
                        set_country_flag = QIN_visible_region_leaders_list_flag
                        set_variable = { QIN_governor_list_page = 1 }
                    }
                }
                else_if = {
                    ROOT = {
                        clr_country_flag = QIN_visible_region_leaders_list_flag
                        set_variable = { QIN_governor_list_page = 0 }
                    }
               }
            }
        }

	}

    QIN_region_leaders_list = {
		context_type = player_context
        parent_window_token = top_bar
		window_name = "QIN_region_leaders_list"

		visible = {
			ROOT = { has_country_flag = QIN_visible_region_leaders_list_flag }
		}

        ai_enabled = {
            always = no
        }

        effects = {
     		close_button_click = {
                ROOT = {
                    clr_country_flag = QIN_visible_region_leaders_list_flag
                    set_variable = { QIN_governor_list_page = 0 }
                }
            }
            open_governor_list_button_click = {
                set_variable = { QIN_governor_list_page = 1 }
            }
            open_viceroys_list_button_click = {
               set_variable = { QIN_governor_list_page = 2 }
            }
        }

        ai_test_interval = 290  # The higher, the long it takes the ai to do actions
        ai_test_variance = 0.5

    }

    QIN_governor_list_page_content = {
		context_type = player_context

		window_name = "QIN_governor_list_page"
		parent_window_name = "QIN_region_leaders_list"

		visible = {
			is_ai = no
			check_variable = { 
				QIN_governor_list_page = 1
			}
		}
	}

    QIN_viceroys_list_page_content = {
		context_type = player_context

		window_name = "QIN_viceroys_list_page"
		parent_window_name = "QIN_region_leaders_list"

		visible = {
			is_ai = no
			check_variable = { 
				QIN_governor_list_page = 2 
			}
		}
	}
