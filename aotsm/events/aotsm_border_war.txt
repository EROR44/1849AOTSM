	id = abw.1 #Win BYZ force
	title = abw.1.t
	desc = abw.1.d
	picture = GFX_report_event_communist_leader
	
	is_triggered_only = yes
	
	option = {
		name = abw.1.a
		trigger = {
			OR = {
				original_tag = SYR
				original_tag = KUR
			}
		}
		add_stability = 0.02
	}
	
	option = {
		name = abw.1.b
		trigger = {
			original_tag = HAL 
		}
		add_stability = -0.02
		if = {
			limit = {
				NOT = {
					HAL = {
						has_country_flag = fed_inv
					}
				}
			}
			HAL = {
				set_country_flag = fed_inv
			}
		}
		if = {
			limit = {
				HAL = {
					has_country_flag = fed_inv
				}
			}
			HAL = {	
				modify_country_flag = {
					flag = fed_inv
					value = 1
				}
			}
		}
	}
}
